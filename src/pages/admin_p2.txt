  const adminExtractResume = async (file: File) => {
    try {
      setIsExtracting(true);
      toast.info('AI Sensing Cluster: Parsing Resume Payload...');
      
      const res = await fetch(`${API_BASE}/admin/extract-resume`, {
        method: 'POST',
        body: file
      });
      
      const data = await res.json();
      if (data.skills) {
         setAiExtracted(data);
         toast.success('Talent Sensing Complete: Merged Data Ready for Audit');
         setModalTab('AI');
      } else {
         toast.error('AI Cluster Analysis Failure');
      }
    } catch (err) {
      toast.error('Resume Sensing failure');
    } finally {
      setIsExtracting(false);
    }
  };
